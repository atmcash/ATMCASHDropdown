  8 var dropdowns = $H();
  9 var Dropdowns = Class.create({
 10     /** @lends Dropdowns# */
 11     /**
 12      * Constructs an ATMCASHDropdown object
 13      * @version 0.4.6
 14      * @author (c) 2010 Ran Grushkowsky
 15      * @requires Prototype 1.6.1.0
 16      * @class Creates a dynamic javascript dropdown from an existing select dropdown
 17      * @param {String} original_select The id of the Select dropdown we are going to replace
 18      * @param {Hash} settings Settings that can configure the dropdown. <i>(currently none are supported)</i>
 19      * @constructs
 20      */
 21     initialize: function(original_select,settings) {
 22         if(original_select.hasClassName('replaced')) return;//avoid duplicates
 23         this.original_select = original_select.identify();//original select id
 24         this.id = this.original_select+"ATMCDropDown"; //dropdown's id
 25         this.typing_buffer = "";
 26         this.buffer_update = new Date();
 27         this.options = $H({
 28             });
 29 
 30         $H(settings || {}).each(function(s){
 31             if (this.options.keys().include(s.key)) { //if the setting provided is valid
 32                 this.options.set(s.key,s.value);
 33             }
 34             else {
 35                 alert('The setting provided \''+s.key+'\' is illegal');
 36             }
 37         }.bind(this));
 38 
 39         //fix for hidden fields
 40         var parents = $A();
 41         original_select.ancestors().each(function(p) {
 42             if (p.getStyle('display') == 'none') {
 43                 parents.push(p);
 44                 p.show();
 45             }
 46         });
 47         //end fix for hidden fields * suggested by Roberto *
 48 
 49         var width = original_select.getWidth();
 50         var classNames = original_select.className;
 51         original_select.addClassName('replaced');
 52         // create list for styling
 53         var div = new Element('div',{
 54             'class':classNames,
 55             'id':this.id
 56         }).insert(new Element('div',{
 57             'class':'t_l'
 58         })).insert(new Element('div',{
 59             'class':'t_r'
 60         })).insert(new Element('div',{
 61             'class':'b_l'
 62         })).insert(new Element('div',{
 63             'class':'b_r'
 64         }));
 65 
 66         var ul = new Element('ul',{
 67             'class':'selectReplacement'
 68         }).setStyle({
 69             'width':width+'px'
 70         });
 71 
 72         original_select.select('option').each(function(t){
 73 
 74             var li = new Element('li');
 75 
 76             li.update(t.text);
 77             li.selIndex = t.index;
 78             li.onclick = function() {
 79                 Dropdowns.getObject(this.up('[class^=divSelectReplacement]').previous('.replaced').identify())._selectMe(this);
 80             };
 81 
 82             if (t.selected) {
 83                 li.addClassName('selected');
 84                 li.onclick = function() {
 85                     var dropdown_obj = Dropdowns.getObject(this.up('[class^=divSelectReplacement]').previous('.replaced').identify());
 86                     $(dropdown_obj.id).up('.dropdown').setStyle({
 87                         'zIndex':2000
 88                     });
 89                     $(dropdown_obj.id).select('.selectReplacement')[0].addClassName('selectOpen').scrollTop = (this.positionedOffset().top-($(dropdown_obj.id).getHeight()/2));
 90 
 91                     this.onclick = function() {
 92                         Dropdowns.getObject(this.up('[class^=divSelectReplacement]').previous('.replaced').identify())._selectMe(this);
 93                     };
 94                 };
 95             }
 96             li.observe('mouseover',function() {
 97                 this.addClassName('hover');
 98             });
 99             li.observe('mouseout',function() {
100                 this.removeClassName("hover");
101             });
102             ul.insert(li);
103         });
104         div.insert(ul);
105         // add the input and the ul
106         original_select.insert({
107             after:new Element('input',{
108                 'type':'text'
109             }).setStyle({
110                 'background':'transparent',
111                 'width':'1px',
112                 'height':'1px',
113                 'border':'none',
114                 'padding':0,
115                 'outline': 'none',
116                 'resize': 'none'
117             }).addClassName('input_tab').observe('focus',function(e){
118                 var select_div = Event.findElement(e).up('.dropdown');
119                 select_div.setStyle({
120                     'zIndex':2000
121                 });
122                 var select_ul = select_div.select('ul.selectReplacement')[0];
123                 select_ul.addClassName('selectOpen');
124                 var select_li = select_ul.select('.selected')[0];
125                 select_ul.scrollTop = (select_li.positionedOffset().top-(select_ul.getHeight()/2));
126                 select_li.onclick = function() {
127                     Dropdowns.getObject(this.up('[class^=divSelectReplacement]').previous('.replaced').identify())._selectMe(this);
128                 };
129             })
130         }).insert({
131             after:div
132         });
133 
134         //fix for hidden fields
135 
136         parents.invoke('hide');
137 
138         //end fix for hidden fields * suggested by Roberto *
139 
140         //add the current object to hash to be referenced back
141         dropdowns.set(this.original_select,this);
142     },
143     /**
144      * Helper Function: Selects the option provided and closes the dropdown.
145      * @param {Element.li} obj Option to be selected
146      * @private
147      */
148     _selectMe:function(obj) {
149         var dropdown_obj = Dropdowns.getObject(obj.up('[class^=divSelectReplacement]').previous('.replaced').identify());
150         $(dropdown_obj.id).select('.selected').each(function(e){
151             e.removeClassName('selected');
152             e.onclick = function() {
153                 dropdown_obj._selectMe(this);
154             };
155         });
156 
157         dropdown_obj._setVal(obj.selIndex);
158         obj.addClassName('selected');
159         dropdown_obj._closeSel(obj);
160     },
161     /**
162      * Helper Function: Updates the value of the original select to the provided paramter.
163      * @param {Int} selIndex Index of option to select
164      * @private
165      */
166     _setVal:function(selIndex) {
167         $(this.original_select).selectedIndex = selIndex;
168     },
169     /**
170      * Helper Function: Closes an open dropdown
171      * @param {Element.li} obj The option whose dropdown we need to close
172      * @private
173      */
174     _closeSel:function(obj) {
175         // close the ul
176         obj.up('.dropdown').setStyle({
177             'zIndex':0
178         });
179         obj.up().scrollTop = 0;
180         obj.up().removeClassName("selectOpen");
181         obj.onclick = function() {
182             this.up('.dropdown').setStyle({
183                 'zIndex':2000
184             });
185             obj.up().addClassName('selectOpen');
186             obj.up().scrollTop = (obj.positionedOffset().top-($(obj.up()).getHeight()/2));
187             this.onclick = function() {
188                 Dropdowns.getObject(this.up('[class^=divSelectReplacement]').previous('.replaced').identify())._selectMe(this);
189             };
190         };
191         if (Object.isFunction($(this.original_select).onblur)) {
192             $(this.original_select).fire("dropdown:blur");
193             $(this.original_select).onblur();
194         }
195     }
196 });
197 
198 /**
199 * This is a getter function for created ATMCASHDropdown object
200 * @addon
201 * @param {String} name The original select's id, also used to create the dropdown.
202 * @return a grid object
203 * @type ATMCASHDropdown Object
204 */
205 Dropdowns.getObject = function(name) {
206     return dropdowns.get(name);
207 };
208 
209 /**
210 * This is a function that sets/selects the index'th option in the dropdown menu.
211 * @addon
212 * @param {String} obj The original select's id, also used to create the dropdown.
213 * @param {Int} index The index of the option we would like to select
214 */
215 Dropdowns.setSelectedIndex = function(obj,index) {
216     var dropdown_obj = Dropdowns.getObject(obj);
217 
218     dropdown_obj._selectMe($(dropdown_obj.id).down('ul').down(index));
219 };
220 
221 Dropdowns.create = function(){
222     $$('select').each(function(e){
223         (new Dropdowns(e));
224     });
225     if (typeof postDropdowns == 'function') {
226         postDropdowns();
227     }
228     Event.observe(document,'click',function(e){
229         var open = $$('.selectOpen');
230         if (open.length !== 0) {
231             open.each(function(o){
232                 var t = o.down('.selected');
233                 if (!t.hasClassName('hover')) {
234                     Dropdowns.getObject(o.up('[class^=divSelectReplacement]').previous('.replaced').identify())._closeSel(t);
235                 }
236             });
237         }
238     });
239     Event.observe(document,'keyup',function(e){
240         var elm;
241         if ($$('.selectOpen').length !== 0) {
242             switch (e.keyCode) {
243                 case Event.KEY_UP:
244                     elm = (Event.findElement(e).hasClassName('input_tab')?Event.findElement(e).previous('[class^=divSelectReplacement]').down('.selectOpen .selected'):Event.findElement(e).down('.selectOpen .selected'));
245                     if (elm.previous()) {
246                         elm.removeClassName("selected");
247                         elm.previous().addClassName('selected');
248                         elm.up('ul.selectOpen').scrollTop = (elm.positionedOffset().top+elm.getHeight()*2)-elm.up('ul.selectOpen').getHeight();
249                     }
250                     Event.stop(e);
251                     break;
252                 case Event.KEY_DOWN:
253                     elm = (Event.findElement(e).hasClassName('input_tab')?Event.findElement(e).previous('[class^=divSelectReplacement]').down('.selectOpen .selected'):Event.findElement(e).down('.selectOpen .selected'));
254                     if (elm.next()) {
255                         elm.removeClassName("selected");
256                         elm.next().addClassName('selected');
257                         elm.up('ul.selectOpen').scrollTop = (elm.positionedOffset().top+elm.getHeight()*2)-elm.up('ul.selectOpen').getHeight();
258                     }
259                     Event.stop(e);
260                     break;
261                 case Event.KEY_RETURN:
262                     Dropdowns.getObject($$('.selectOpen')[0].up('[class^=divSelectReplacement]').previous('.replaced').identify())._selectMe($$('.selectOpen .selected')[0]);
263                     Event.stop(e);
264                     break;
265                 case Event.KEY_ESC:
266                     Dropdowns.getObject($$('.selectOpen')[0].up('[class^=divSelectReplacement]').previous('.replaced').identify())._closeSel($$('.selectOpen .selected')[0]);
267                     Event.stop(e);
268                     break;
269                 case Event.KEY_TAB:
270                     Event.stop(e);
271                     var other_dropdowns = $A();
272 
273                     if (Event.findElement(e).hasClassName('input_tab')) {
274                         other_dropdowns = $$('.selectOpen').without(Event.findElement(e).previous('[class^=divSelectReplacement]').select('ul.selectOpen')[0]);
275                     }
276                     else {
277                         other_dropdowns = $$('.selectOpen');
278                     }
279                     if (other_dropdowns.size() !== 0) {
280                         Dropdowns.getObject(other_dropdowns[0].up('[class^=divSelectReplacement]').previous('.replaced').identify())._closeSel($$('.selectOpen .selected')[0]);
281                     }
282 
283 
284                     break;
285                 default:
286                     var select = $$('.selectOpen');
287                     if (select.size() > 0) {
288 
289                         select = select[0];
290 
291                         var select_obj = Dropdowns.getObject(select.up('[class^=divSelectReplacement]').previous('.replaced').identify());
292                         //clear buffer if too old
293                         if ((select_obj.buffer_update.getTime() + 1000) < new Date().getTime()) {
294                             select_obj.typing_buffer = "";
295                         }
296                         select_obj.buffer_update = new Date();
297                         select_obj.typing_buffer += String.fromCharCode(e.keyCode).toUpperCase();
298                         var seen_selected = false;
299                         var first_element_seen = null;
300                         var updated = false;
301                         select.descendants().each(function (e){
302                             if (e.innerHTML.toUpperCase().startsWith(select_obj.typing_buffer)) {
303                                 if (first_element_seen == null) {
304                                     first_element_seen = e;
305                                 }
306                                 if (seen_selected) {
307                                     e.up().down('.selected').removeClassName("selected");
308                                     e.addClassName('selected');
309                                     e.up('ul.selectOpen').scrollTop = (e.positionedOffset().top+e.getHeight()*2)-e.up('ul.selectOpen').getHeight();
310                                     updated = true;
311                                     throw $break;
312                                 }
313                             }
314                             if (e.hasClassName('selected')) {
315                                 seen_selected = true;
316                             }
317                         });
318                         if (!updated) {
319                             //if we got here, nothing was selected, mark the first element as selected
320                             if (first_element_seen != null) {
321                                 select.down('.selected').removeClassName("selected");
322                                 first_element_seen.addClassName('selected');
323                                 first_element_seen.up('ul.selectOpen').scrollTop = (first_element_seen.positionedOffset().top+first_element_seen.getHeight()*2)-first_element_seen.up('ul.selectOpen').getHeight();
324                             }
325                         }
326 
327                     }
328             }
329         }
330     });
331 }
332 
333 document.observe("dom:loaded", function() {
334     Dropdowns.create();
335 });
